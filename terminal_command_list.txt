/// ディレクトリを移動
$ cd (任意のディレクトリのパス)

/// リポジトリを初期化
$ git init 

/// リポジトリの状態を確認
$ git status 

/// ステージ領域にファイルを追加(git add .(ワイルドカード)でディレクトリの全てのファイルを指定)
$ git add (ファイル名)

/// リポジトリの歴史を記録
$ git commit -m "First commit"

/// コミットログを確認(git log (ディレクトリ名・ファイル名)で特定のログのみを表示する)
$ git log 

/// ファイル差分を表示する
$ git log -p
$ git log -p README.md

/// 変更差分を確認(ワークツリーとステージ領域の差分を確認)
$ git diff

/// ブランチを一覧表示(*がついているのが現在のブランチ)
$ git branch

/// ブランチを作成し切り替える
$ git checkout -b (ブランチ名)

/// ブランチを切り替える
$ git checkout (ブランチ名)

/// ブランチをマージ(--no-ffでマージコミットの作成)
$ git merge --no-ff (取り込みブランチ)

/// ブランチを視覚的に確認する
$ git log --graph

/// ディレクトリを作成
$ mkdir (作成するディレクトリ名)
/// 複数階層のディレクトリを作成
$ mkdir -p summary/2025/01/25

/// ファイルの新規作成(既存のファイルと同名のものを作成しても上書きや削除はされない)
$ touch (新しいファイル名1) (新しいファイル名2)

/// ファイルやディレクトリの削除
$ rm -r (削除するファイル名1) (削除するファイル名2)
/// ファイルやディレクトリの削除(削除前の確認つき -i yes noを入力)
$ rm -i (削除するファイル名1) (削除するファイル名2) 

/// ファイルの内容表示(-n で行番号表示)git 
$ cat -n (ファイル名1) (ファイル名2) (ファイル名3)
